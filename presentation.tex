	\begin{frame}{Theory - Bayesian Statistics}
		Based on Bayers Theorem
		\begin{equation}
			P(A|B) = \frac{P(B|A) P(A)}{P(B)}
		\end{equation}
		Consider the case
		\begin{align*}
			P(A|B) &\propto P(B|A) P(A)
		\end{align*}
		"Posterior is proportional to the likelihood times the prior"
		
		Formal Construction
		\begin{equation*}
			\pi (\theta | x) = \mathcal{L} (x | \theta) \cdot 
				\frac{\pi(\theta)}{\int_\Theta \mathcal{L} (x | \theta') \cdot \pi (\theta') d \theta'}
		\end{equation*}
	\end{frame}
