	\begin{frame}{Theory - Bayesian Statistics}
		Based on Bayes' Theorem
		\begin{equation}
			P(A|B) = \frac{P(B|A) P(A)}{P(B)}
		\end{equation}

		A $\rightarrow$ Parameters of the model

		B $\rightarrow$ Observed gravitational waves

		As the measurements have been made, P(B) is fixed, and we can consider the proportionality form of Bayes' theorm,
		\begin{align*}
			P(A|B) &\propto P(B|A) P(A)
		\end{align*}
		"Posterior is proportional to the likelihood times the prior"
